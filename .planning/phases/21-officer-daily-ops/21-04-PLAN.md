     ```typescript
     const bulkApprove = useMutation({
       ...bulkApproveMembershipApplicationsMutation(),
       onSuccess: (data) => {
         const { succeeded, failed } = data;
         if (failed.length === 0) {
           toast.success(`${succeeded.length} applications approved`);
         } else if (succeeded.length === 0) {
           toast.error(`All ${failed.length} approvals failed`);
         } else {
           toast.warning(`${succeeded.length} approved, ${failed.length} failed`);
         }
         // Show individual failures
         failed.forEach((f: any) => toast.error(`${f.id.slice(0,8)}...: ${f.reason}`));
         setSelectedIds(new Set());
         invalidateApplications();
       },
       onError: () => toast.error('Bulk approve failed'),
     });
     ```
