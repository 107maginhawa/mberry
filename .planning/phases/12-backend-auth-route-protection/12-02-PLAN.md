```typescript
export function requireOrgRole(ctx: BaseContext, allowedRoles: readonly string[]): Response | null {
  const membership = ctx.get('orgMembership');
  if (!membership) return ctx.json({ error: 'Organization membership required' }, 403);
  if (!allowedRoles.includes(membership.role)) return ctx.json({ error: `Requires one of: ${allowedRoles.join(', ')}` }, 403);
  return null;
}
```

```typescript
import { describe, test, expect, beforeAll } from 'bun:test';
import { apiAs, type ApiClient } from '@/tests/helpers/api-as';

describe('Association mutation routes - officer protection', () => {
  let memberClient: ApiClient;
  
  beforeAll(async () => {
    memberClient = await apiAs('member@memberry.ph');
  });

  const ORG_ID = 'ed8e3a96-8126-4341-be42-e6eb7940c562'; // pda-metro-manila from seed

  // Mutation routes that should require officer access
  const officerMutationRoutes = [
    { method: 'POST', path: '/association/events', body: { title: 'Test', organizationId: ORG_ID, startDate: '2026-06-01', endDate: '2026-06-02' }, label: 'create event' },
    { method: 'PATCH', path: '/association/events/fake-id', body: { title: 'Updated' }, label: 'update event' },
    { method: 'DELETE', path: '/association/events/fake-id', label: 'delete event' },
    { method: 'POST', path: '/association/events/fake-id/cancel', label: 'cancel event' },
    { method: 'POST', path: '/association/events/fake-id/publish', label: 'publish event' },
    { method: 'POST', path: '/association/events/checkins', body: { eventId: 'fake-id', personId: 'fake-id' }, label: 'check in' },
    { method: 'POST', path: '/association/training', body: { title: 'Test', organizationId: ORG_ID }, label: 'create training' },
    { method: 'PATCH', path: '/association/training/fake-id', body: { title: 'Updated' }, label: 'update training' },
    { method: 'POST', path: '/association/member/elections', body: { organizationId: ORG_ID, title: 'Test' }, label: 'create election' },
    { method: 'POST', path: '/association/member/dues-configs', body: { organizationId: ORG_ID }, label: 'create dues config' },
    { method: 'POST', path: '/association/member/dues-invoices/generate', body: { organizationId: ORG_ID }, label: 'generate invoices' },
    { method: 'POST', path: '/association/member/dues-payments', body: { amount: 100 }, label: 'record payment' },
    { method: 'POST', path: '/association/member/dues-payments/fake-id/refund', body: { reason: 'test' }, label: 'refund payment' },
    { method: 'POST', path: '/association/member/memberships', body: { organizationId: ORG_ID, personId: 'fake-id' }, label: 'create membership' },
    { method: 'PATCH', path: '/association/member/memberships/fake-id', body: { status: 'active' }, label: 'update membership' },
    { method: 'POST', path: '/association/member/officer-terms', body: { organizationId: ORG_ID }, label: 'assign officer' },
    { method: 'POST', path: `/association/communications/announcements/${ORG_ID}`, body: { subject: 'Test' }, label: 'create announcement' },
  ];

  for (const route of officerMutationRoutes) {
    test(`member blocked: ${route.label} (${route.method} ${route.path}) returns 403`, async () => {
      let res: Response;
      const path = route.path;
      if (route.method === 'POST') {
        res = await memberClient.post(path, route.body || {});
      } else if (route.method === 'PATCH') {
        res = await memberClient.patch(path, route.body || {});
      } else if (route.method === 'DELETE') {
        res = await memberClient.delete(path);
      } else {
        res = await memberClient.get(path);
      }
      expect(res.status).toBe(403);
    });
  }

  // Member-accessible GET routes (per D-07)
  test('member can GET /association/events (read-only)', async () => {
    const res = await memberClient.get(`/association/events?organizationId=${ORG_ID}`);
    expect(res.status).not.toBe(403);
  });
});
```
