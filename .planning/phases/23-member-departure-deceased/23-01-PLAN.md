   Do NOT add `lapsed` -- it already exists (Pitfall 3 from RESEARCH.md).

2. Add two new request models AFTER `MembershipTerminateRequest`:
   ```typespec
   @doc("Request to record a member's voluntary resignation")
   model MembershipResignRequest {
     @doc("Reason for resignation")
     @maxLength(500)
     terminationReason?: string;
   }

   @doc("Request to record a member's death")
   model MembershipDeceasedRequest {
     @doc("Date of death")
     dateOfDeath: plainDate;

     @doc("Optional notes")
     @maxLength(500)
     terminationReason?: string;
   }
   ```
   Note: `terminationReason` is optional on resign (unlike terminate which requires it). `dateOfDeath` is required on deceased.

3. Add two new operations inside the `MembershipManagement` interface, AFTER `terminateMembership`:
   ```typespec
   @doc("Record a member's voluntary resignation.")
   @operationId("resignMembership")
   @post
   @route("/{membershipId}/resign")
   @useAuth(bearerAuth)
   @extension("x-security-required-roles", #["association:admin"])
   resignMembership(
     @path membershipId: string,
     @body body: MembershipResignRequest
   ): ApiOkResponse<Membership>
     | ApiBadRequestResponse
     | ApiNotFoundResponse
     | ApiConflictResponse
     | ApiForbiddenResponse;

   @doc("Record a member's death.")
   @operationId("deceaseMembership")
   @post
   @route("/{membershipId}/deceased")
   @useAuth(bearerAuth)
   @extension("x-security-required-roles", #["association:admin"])
   deceaseMembership(
     @path membershipId: string,
     @body body: MembershipDeceasedRequest
   ): ApiOkResponse<Membership>
     | ApiBadRequestResponse
     | ApiNotFoundResponse
     | ApiConflictResponse
     | ApiForbiddenResponse;
   ```
