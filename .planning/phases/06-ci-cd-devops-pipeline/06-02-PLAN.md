Create `.github/workflows/deploy.yml` with the following structure:

```yaml
name: Deploy

on:
  push:
    branches: [main]
  workflow_dispatch:
    inputs:
      environment:
        description: 'Deploy target'
        required: true
        type: choice
        options:
          - staging
          - production
```

**Job: build**
- runs-on: ubuntu-latest
- timeout-minutes: 15
- outputs: image_tag (the full ghcr.io tag with SHA)
- Steps:
  1. actions/checkout@v4
  2. docker/setup-buildx-action@v3
  3. docker/login-action@v3 with registry: ghcr.io, username: ${{ github.actor }}, password: ${{ secrets.GITHUB_TOKEN }}
  4. docker/build-push-action@v6:
     - context: .
     - file: services/api-ts/Dockerfile
     - push: true
     - tags: `ghcr.io/${{ github.repository }}/api:${{ github.sha }}` and `ghcr.io/${{ github.repository }}/api:latest`
     - cache-from: type=gha
     - cache-to: type=gha,mode=max
  5. Set output: `echo "image_tag=ghcr.io/${{ github.repository }}/api:${{ github.sha }}" >> $GITHUB_OUTPUT`
  6. Build frontend apps (same pattern as ci.yml build-frontends but with VITE_API_URL set per environment):
     - oven-sh/setup-bun@v2 with bun-version: 1.2.21
     - bun install --frozen-lockfile
     - Build OpenAPI spec + generate codegen
     - Build all 3 frontend apps with `VITE_API_URL=${{ vars.STAGING_API_URL || '' }}`
     - Upload dist/ artifacts for each app (actions/upload-artifact@v4)

**Job: deploy-staging**
- if: `github.event_name == 'push' || (github.event_name == 'workflow_dispatch' && inputs.environment == 'staging')`
- needs: [build]
- runs-on: ubuntu-latest
- timeout-minutes: 10
- environment: staging
- Steps:
  1. actions/checkout@v4
  2. Download frontend artifacts (actions/download-artifact@v4)
  3. Deploy API placeholder step:
     ```yaml
     - name: Deploy API to staging
       run: |
         echo "::notice::Deploy API image ${{ needs.build.outputs.image_tag }} to staging"
         echo "TODO: Add platform-specific deploy command here"
         echo "Example: railway up --image ${{ needs.build.outputs.image_tag }}"
         echo "Example: fly deploy --image ${{ needs.build.outputs.image_tag }}"
     ```
  4. Deploy frontends placeholder step:
     ```yaml
     - name: Deploy frontends to staging
       run: |
         echo "::notice::Deploy frontend artifacts to staging"
         echo "TODO: Add platform-specific deploy command here"
         echo "Example: wrangler pages deploy apps/memberry/dist/ --project-name memberry-staging"
     ```
  5. Health check gate:
     ```yaml
     - name: Health check — staging API
       if: vars.STAGING_API_URL != ''
       run: |
         DEPLOY_URL="${{ vars.STAGING_API_URL }}"
         for i in $(seq 1 30); do
           STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$DEPLOY_URL/readyz")
           echo "Attempt $i: HTTP $STATUS"
           [ "$STATUS" = "200" ] && echo "Staging API ready" && exit 0
           sleep 5
         done
         echo "::error::Staging health check failed after 150s"
         exit 1
     ```

The deploy steps are platform-agnostic placeholders. The user configures the actual deploy commands when they choose a platform (Railway, Fly, Render, etc). The health check gate is real and will work once STAGING_API_URL is set.

Do NOT hardcode any platform-specific commands. Use echo/notice steps so the workflow is valid YAML and runs green even without a platform configured.

**Job: deploy-production**
- if: `github.event_name == 'workflow_dispatch' && inputs.environment == 'production'`
- needs: [build]
- runs-on: ubuntu-latest
- timeout-minutes: 15
- environment: production (GitHub Environment with required reviewers protects this)
- Steps:
  1. actions/checkout@v4
  2. Download frontend artifacts (actions/download-artifact@v4)
  3. Deploy API placeholder:
     ```yaml
     - name: Deploy API to production
       run: |
         echo "::notice::Deploy API image ${{ needs.build.outputs.image_tag }} to production"
         echo "TODO: Add platform-specific deploy command here"
         echo "Uses SAME image SHA that was validated in staging"
     ```
  4. Deploy frontends placeholder (same pattern as staging, different env vars)
  5. Health check gate (stricter — longer timeout, retry logic):
     ```yaml
     - name: Health check — production API
       if: vars.PRODUCTION_API_URL != ''
       run: |
         DEPLOY_URL="${{ vars.PRODUCTION_API_URL }}"
         for i in $(seq 1 30); do
           STATUS=$(curl -s -o /dev/null -w "%{http_code}" "$DEPLOY_URL/readyz")
           echo "Attempt $i: HTTP $STATUS"
           if [ "$STATUS" = "200" ]; then
             echo "Production API ready"
             # Double-check: verify /readyz returns healthy JSON
             curl -s "$DEPLOY_URL/readyz" | grep -q '"status":"ok"' || {
               echo "::warning::API responded 200 but readyz body unexpected"
             }
             exit 0
           fi
           sleep 5
         done
         echo "::error::Production health check failed after 150s"
         exit 1
     ```
  6. Post-deploy smoke test:
     ```yaml
     - name: Smoke test — production
       if: vars.PRODUCTION_API_URL != ''
       run: |
         # Verify key endpoints respond
         curl -fsS "${{ vars.PRODUCTION_API_URL }}/livez" || exit 1
         echo "Production smoke test passed"
     ```

Production deploy is ONLY triggered via workflow_dispatch with environment=production. The GitHub Environment protection rules (required reviewers) provide the approval gate. The build job reuses the same image SHA — no rebuild.
