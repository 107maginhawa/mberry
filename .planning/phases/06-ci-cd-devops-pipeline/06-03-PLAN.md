Create `.github/workflows/monitor.yml`:

```yaml
name: Production Health Monitor

on:
  schedule:
    - cron: '*/5 * * * *'  # Every 5 minutes
  workflow_dispatch:  # Allow manual trigger for testing

jobs:
  health-ping:
    runs-on: ubuntu-latest
    timeout-minutes: 2
    if: vars.PRODUCTION_API_URL != ''
    steps:
      - name: Check /readyz
        id: health
        run: |
          URL="${{ vars.PRODUCTION_API_URL }}/readyz"
          STATUS=$(curl -s -o /tmp/readyz-body.txt -w "%{http_code}" --max-time 10 "$URL" || echo "000")
          echo "status=$STATUS" >> $GITHUB_OUTPUT
          echo "Health check: HTTP $STATUS"
          if [ "$STATUS" = "200" ]; then
            echo "Production API healthy"
            cat /tmp/readyz-body.txt
          else
            echo "::error::Production API unhealthy (HTTP $STATUS)"
            cat /tmp/readyz-body.txt 2>/dev/null || echo "No response body"
          fi

      - name: Check /livez
        id: liveness
        if: steps.health.outputs.status != '200'
        run: |
          URL="${{ vars.PRODUCTION_API_URL }}/livez"
          STATUS=$(curl -s -o /dev/null -w "%{http_code}" --max-time 10 "$URL" || echo "000")
          echo "liveness=$STATUS" >> $GITHUB_OUTPUT
          echo "Liveness check: HTTP $STATUS"

      - name: Check for existing incident issue
        if: steps.health.outputs.status != '200'
        id: existing
        uses: actions/github-script@v7
        with:
          script: |
            const issues = await github.rest.issues.listForRepo({
              owner: context.repo.owner,
              repo: context.repo.repo,
              labels: 'incident,production',
              state: 'open',
              per_page: 1
            });
            const exists = issues.data.length > 0;
            core.setOutput('has_open_incident', exists.toString());
            if (exists) {
              core.info(`Open incident issue exists: #${issues.data[0].number}`);
              // Add a comment to the existing issue with latest status
              await github.rest.issues.createComment({
                owner: context.repo.owner,
                repo: context.repo.repo,
                issue_number: issues.data[0].number,
                body: `Health check still failing.\n- /readyz: HTTP ${process.env.HEALTH_STATUS}\n- /livez: HTTP ${process.env.LIVENESS_STATUS || 'not checked'}\n- Time: ${new Date().toISOString()}`
              });
            }
          env:
            HEALTH_STATUS: ${{ steps.health.outputs.status }}
            LIVENESS_STATUS: ${{ steps.liveness.outputs.liveness }}

      - name: Create incident issue
        if: steps.health.outputs.status != '200' && steps.existing.outputs.has_open_incident == 'false'
        uses: actions/github-script@v7
        with:
          script: |
            const healthStatus = process.env.HEALTH_STATUS;
            const livenessStatus = process.env.LIVENESS_STATUS || 'not checked';
            const apiUrl = process.env.API_URL;

            await github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: `[INCIDENT] Production API unhealthy — /readyz HTTP ${healthStatus}`,
              body: [
                '## Production Health Check Failure',
                '',
                `**Time:** ${new Date().toISOString()}`,
                `**API URL:** ${apiUrl}`,
                `**/readyz status:** HTTP ${healthStatus}`,
                `**/livez status:** HTTP ${livenessStatus}`,
                '',
                '### Diagnosis',
                livenessStatus === '200'
                  ? '- Process is alive (/livez OK) but not ready (/readyz failing) — likely a dependency issue (DB, storage, or jobs)'
                  : '- Process may be down or unreachable — check deployment logs',
                '',
                '### Resolution',
                '1. Check deployment platform logs',
                '2. Verify DATABASE_URL connectivity',
                '3. Verify storage (MinIO/S3) connectivity',
                '4. Close this issue when resolved — monitor will stop adding comments',
                '',
                '*Auto-generated by production health monitor*'
              ].join('\n'),
              labels: ['incident', 'production']
            });
          env:
            HEALTH_STATUS: ${{ steps.health.outputs.status }}
            LIVENESS_STATUS: ${{ steps.liveness.outputs.liveness }}
            API_URL: ${{ vars.PRODUCTION_API_URL }}

      - name: Auto-close resolved incidents
        if: steps.health.outputs.status == '200'
        uses: actions/github-script@v7
        with:
          script: |
            const issues = await github.rest.issues.listForRepo({
              owner: context.repo.owner,
              repo: context.repo.repo,
              labels: 'incident,production',
              state: 'open',
              per_page: 5
            });
            for (const issue of issues.data) {
              await github.rest.issues.createComment({
                owner: context.repo.owner,
                repo: context.repo.repo,
                issue_number: issue.number,
                body: `Resolved. /readyz returned HTTP 200 at ${new Date().toISOString()}.\n\nAuto-closing.`
              });
              await github.rest.issues.update({
                owner: context.repo.owner,
                repo: context.repo.repo,
                issue_number: issue.number,
                state: 'closed'
              });
            }
```

Key design decisions:
- Checks /readyz first. If unhealthy, also checks /livez to provide diagnostic info (alive but not ready vs completely down).
- Deduplicates: searches for open issues with labels `incident` + `production`. If one exists, adds a comment instead of creating a new issue.
- Auto-closes: when health recovers, finds and closes open incident issues with a resolution comment.
- The `if: vars.PRODUCTION_API_URL != ''` guard means the workflow is a no-op until the variable is configured — safe to merge before prod exists.
- workflow_dispatch allows manual testing of the monitor.

Also ensure the repo has the `incident` and `production` labels. Add a note in the workflow comments that these labels should exist (GitHub will auto-create them on first issue creation if they don't, but with default colors).
