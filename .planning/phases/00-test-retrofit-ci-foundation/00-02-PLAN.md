Structure with 3 parallel jobs:

```yaml
name: CI
on:
  push:
    branches: [main]
  pull_request:

jobs:
  lint-typecheck:
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: 1.2.21
      - run: bun install --frozen-lockfile
      - name: Build OpenAPI spec
        run: bun --filter @monobase/api-spec run build
      - name: Generate API codegen
        run: bun --filter @monobase/api-ts run generate
      - name: Typecheck
        run: bun run typecheck
      - name: Lint
        run: bun run lint

  e2e:
    runs-on: ubuntu-latest
    timeout-minutes: 20
    services:
      postgres:
        image: postgres:16-alpine
        env:
          POSTGRES_USER: postgres
          POSTGRES_PASSWORD: password
          POSTGRES_DB: monobase
        ports: ['5432:5432']
        options: >-
          --health-cmd "pg_isready -U postgres -d monobase"
          --health-interval 5s --health-timeout 5s --health-retries 10
      minio:
        image: minio/minio:latest
        ports: ['9000:9000']
        env:
          MINIO_ROOT_USER: minioadmin
          MINIO_ROOT_PASSWORD: minioadmin
        options: >-
          --health-cmd "curl -fsS http://localhost:9000/minio/health/live"
          --health-interval 5s --health-timeout 5s --health-retries 10
    env:
      DATABASE_URL: postgresql://postgres:password@localhost:5432/monobase
      PORT: 7213
      API_URL: http://localhost:7213
      AUTH_SECRET: contract-test-secret-do-not-use-in-prod
      STORAGE_PROVIDER: minio
      STORAGE_ENDPOINT: http://localhost:9000
      STORAGE_BUCKET: monobase-files
      STORAGE_ACCESS_KEY_ID: minioadmin
      STORAGE_SECRET_ACCESS_KEY: minioadmin
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: 1.2.21
      - run: bun install --frozen-lockfile
      - name: Build OpenAPI spec
        run: bun --filter @monobase/api-spec run build
      - name: Generate API codegen
        run: bun --filter @monobase/api-ts run generate
      - name: Install Playwright browsers
        run: cd apps/memberry && bunx playwright install --with-deps chromium
      - name: Boot API
        run: |
          cd services/api-ts
          bun dev > /tmp/api.log 2>&1 &
          echo $! > /tmp/api.pid
      - name: Wait for API
        run: |
          for i in $(seq 1 60); do
            curl -fsS "$API_URL/livez" && exit 0
            sleep 1
          done
          echo "API never became ready"
          tail -200 /tmp/api.log
          exit 1
      - name: Seed database
        run: cd services/api-ts && bun run db:seed-modules
      - name: Start Memberry app
        run: |
          cd apps/memberry
          bun dev > /tmp/memberry.log 2>&1 &
          echo $! > /tmp/memberry.pid
      - name: Wait for Memberry app
        run: |
          for i in $(seq 1 30); do
            curl -fsS "http://localhost:3004" > /dev/null 2>&1 && exit 0
            sleep 1
          done
          echo "Memberry app never became ready"
          tail -100 /tmp/memberry.log
          exit 1
      - name: Run E2E tests
        run: cd apps/memberry && bun run test:e2e
      - name: Upload test results
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: playwright-report
          path: apps/memberry/playwright-report/

  contract:
    runs-on: ubuntu-latest
    timeout-minutes: 15
    services:
      postgres:
        image: postgres:16-alpine
        env:
          POSTGRES_USER: postgres
          POSTGRES_PASSWORD: password
          POSTGRES_DB: monobase
        ports: ['5432:5432']
        options: >-
          --health-cmd "pg_isready -U postgres -d monobase"
          --health-interval 5s --health-timeout 5s --health-retries 10
      minio:
        image: minio/minio:latest
        ports: ['9000:9000']
        env:
          MINIO_ROOT_USER: minioadmin
          MINIO_ROOT_PASSWORD: minioadmin
        options: >-
          --health-cmd "curl -fsS http://localhost:9000/minio/health/live"
          --health-interval 5s --health-timeout 5s --health-retries 10
    env:
      DATABASE_URL: postgresql://postgres:password@localhost:5432/monobase
      PORT: 7213
      API_URL: http://localhost:7213
      AUTH_SECRET: contract-test-secret-do-not-use-in-prod
      STORAGE_PROVIDER: minio
      STORAGE_ENDPOINT: http://localhost:9000
      STORAGE_BUCKET: monobase-files
      STORAGE_ACCESS_KEY_ID: minioadmin
      STORAGE_SECRET_ACCESS_KEY: minioadmin
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: 1.2.21
      - run: bun install --frozen-lockfile
      - name: Build OpenAPI spec
        run: bun --filter @monobase/api-spec run build
      - name: Generate API codegen
        run: bun --filter @monobase/api-ts run generate
      - name: Install Hurl
        run: |
          curl -sSL https://github.com/Orange-OpenSource/hurl/releases/download/6.0.0/hurl_6.0.0_amd64.deb -o /tmp/hurl.deb
          sudo dpkg -i /tmp/hurl.deb
      - name: Boot API
        run: |
          cd services/api-ts
          bun dev > /tmp/api.log 2>&1 &
          echo $! > /tmp/api.pid
      - name: Wait for API
        run: |
          for i in $(seq 1 60); do
            curl -fsS "$API_URL/livez" && exit 0
            sleep 1
          done
          echo "API never became ready"
          tail -200 /tmp/api.log
          exit 1
      - name: Seed database
        run: cd services/api-ts && bun run db:seed-modules
      - name: Run contract tests
        run: hurl --test --variable api=$API_URL specs/api/tests/contract/*.hurl
```

# Requires: seeded org ed8e3a96-8126-4341-be42-e6eb7940c562

{

session_token: cookie "better-auth.session_token"

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions -> Secrets | CI uses hardcoded test-only secrets, no production credentials |
| CI -> Services | Postgres/MinIO provisioned with test-only creds in ephemeral containers |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-00-03 | I (Information Disclosure) | CI env vars | mitigate | All secrets in ci.yml are test-only values (AUTH_SECRET: contract-test-secret-do-not-use-in-prod). No production secrets stored in workflow file. Production secrets use GitHub Secrets if needed later. |
| T-00-04 | T (Tampering) | PR-triggered CI | accept | Standard GitHub Actions security model — PRs from forks have read-only access, no secrets exposure. Acceptable for open-source CI. |
| T-00-05 | E (Elevation of Privilege) | Hurl test auth | accept | Tests use fixed test credentials against ephemeral test database. No privilege escalation path to production. |
</threat_model>
