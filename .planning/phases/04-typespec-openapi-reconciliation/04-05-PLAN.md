<interfaces>
<!-- Migration pattern: manual fetch → generated SDK hook -->
BEFORE:
```typescript
import { useQuery } from '@tanstack/react-query'
import { api } from '@/lib/api'
const { data } = useQuery({
  queryKey: ['my-certificates'],
  queryFn: async () => api.get<{ data: any[] }>('/api/certificates/my'),
})
```

AFTER:
```typescript
import { useQuery } from '@tanstack/react-query'
import { listMyCertificatesOptions } from '@monobase/sdk-ts/generated/react-query'
const { data } = useQuery(listMyCertificatesOptions())
```

BEFORE (mutation):
```typescript
import { useMutation } from '@tanstack/react-query'
import { api } from '@/lib/api'
const mutation = useMutation({
  mutationFn: (body) => api.post('/api/elections/create/' + orgId, body),
})
```

AFTER (mutation):
```typescript
import { useMutation } from '@tanstack/react-query'
import { createElectionMutation } from '@monobase/sdk-ts/generated/react-query'
const mutation = useMutation(createElectionMutation())
```

<!-- Hook naming convention -->
- List (GET collection): {operationId}Options → listElectionsOptions
- Get (GET single): {operationId}Options → getElectionOptions
- Create/Update/Delete: {operationId}Mutation → createElectionMutation
- Exact names depend on @operationId in TypeSpec — verify in react-query.gen.ts
</interfaces>
</context>
