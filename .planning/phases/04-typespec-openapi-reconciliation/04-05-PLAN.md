<!-- Migration pattern: manual fetch -> generated SDK hook -->
BEFORE:
```typescript
import { useQuery } from '@tanstack/react-query'
import { api } from '@/lib/api'
const { data } = useQuery({
  queryKey: ['dues-payments'],
  queryFn: async () => api.get<{ data: any[] }>('/api/dues/payments'),
})
```

AFTER:
```typescript
import { useQuery } from '@tanstack/react-query'
import { listDuesPaymentsOptions } from '@monobase/sdk-ts/generated/react-query'
const { data } = useQuery(listDuesPaymentsOptions())
```

BEFORE (mutation):
```typescript
import { useMutation } from '@tanstack/react-query'
import { api } from '@/lib/api'
const mutation = useMutation({
  mutationFn: (body) => api.post('/api/dues/payments', body),
})
```

AFTER (mutation):
```typescript
import { useMutation } from '@tanstack/react-query'
import { recordDuesPaymentMutation } from '@monobase/sdk-ts/generated/react-query'
const mutation = useMutation(recordDuesPaymentMutation())
```

<!-- Hook naming convention -->
- List (GET collection): {operationId}Options
- Get (GET single): {operationId}Options
- Create/Update/Delete: {operationId}Mutation
- Exact names depend on @operationId in TypeSpec — verify in react-query.gen.ts
</interfaces>
