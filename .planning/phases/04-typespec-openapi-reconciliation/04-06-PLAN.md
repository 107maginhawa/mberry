```typescript
import { useQuery } from '@tanstack/react-query'
import { api } from '@/lib/api'
const { data } = useQuery({
```

```typescript

import { listMyCertificatesOptions } from '@monobase/sdk-ts/generated/react-query'

```

- Create/Update/Delete: {operationId}Mutation
